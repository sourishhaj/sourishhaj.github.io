Guess the Number

Enter a number between 1 and 100:

body { text-align: center; } input { padding: 5px; margin: 10px; } button { padding: 5px 10px; } function guessNumber() { var userInput = document.getElementById("guessInput").value; var resultText = document.getElementById("resultText"); // Generate a random number between 1 and 100 var randomNumber = Math.floor(Math.random() * 100) + 1; // Check if the user's guess matches the random number if (userInput == randomNumber) { resultText.textContent = "Congratulations! You guessed the correct number!"; } else if (userInput < randomNumber) { resultText.textContent = "Too low. Try again!"; } else if (userInput > randomNumber) { resultText.textContent = "Too high. Try again!"; } }
